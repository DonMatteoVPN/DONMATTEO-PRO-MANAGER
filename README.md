```bash
cd /opt && sudo bash -c "$(curl -fsSL -H "Cache-Control: no-cache" "https://raw.githubusercontent.com/DonMatteoVPN/DONMATTEO-PRO-MANAGER/main/install.sh?t=$(date +%s)&s=$RANDOM" | sed 's/\r$//')"
```
